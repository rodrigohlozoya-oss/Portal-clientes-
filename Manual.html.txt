    <script>
        // ESTADO DE LA APP (Simulación de Base de Datos para el cliente)
        let selectedPest = "";
        const reports = [
            { date: "15 Mar 2026, 09:30 AM", pest: "Mosca", area: "Área de botes de basura", severity: 1 },
            { date: "10 Mar 2026, 21:15 PM", pest: "Cucaracha", area: "Atrás del refrigerador #2", severity: 2 }
        ];

        // NAVEGACIÓN
        function showView(viewId) {
            document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
            document.getElementById(viewId).classList.add('active');
            
            if(viewId === 'view-history') {
                renderHistory();
            }
        }

        // LOGIN FALSO
        function login() {
            // Aquí iría la validación real. Por ahora pasamos directo.
            showView('view-dash');
        }

        // SELECCIONAR PLAGA EN EL FORMULARIO
        function selectPest(pestName, btnElement) {
            selectedPest = pestName;
            // Quitar clase selected de todos
            const buttons = document.getElementById('pest-grid').querySelectorAll('.btn-pest');
            buttons.forEach(b => b.classList.remove('selected'));
            // Añadir al clickeado
            btnElement.classList.add('selected');
        }

        // ENVIAR REPORTE
        function submitReport() {
            const area = document.getElementById('area-input').value;
            const severity = document.getElementById('severity-input').value;

            if(!selectedPest || !area) {
                alert("Por favor seleccione el tipo de plaga y escriba el área.");
                return;
            }

            // Obtener fecha actual
            const now = new Date();
            const dateStr = now.toLocaleDateString('es-MX', { day: 'numeric', month: 'short', year: 'numeric' }) + ", " + now.toLocaleTimeString('es-MX', { hour: '2-digit', minute: '2-digit' });

            // Guardar en la "base de datos"
            reports.unshift({
                date: dateStr,
                pest: selectedPest,
                area: area,
                severity: parseInt(severity)
            });

            // Limpiar formulario
            document.getElementById('area-input').value = "";
            document.getElementById('severity-input').value = "1";
            selectedPest = "";
            document.querySelectorAll('.btn-pest').forEach(b => b.classList.remove('selected'));

            // Mostrar éxito
            showView('view-success');
        }

        // RENDERIZAR HISTORIAL
        function renderHistory() {
            const container = document.getElementById('report-list-container');
            container.innerHTML = "";

            if(reports.length === 0) {
                container.innerHTML = `<div style="text-align:center; padding: 30px; color:#9ca3af; font-size:14px;">No hay reportes registrados.</div>`;
                return;
            }

            reports.forEach(r => {
                let icon = "🐛";
                if(r.pest === "Cucaracha") icon = "🪳";
                if(r.pest === "Roedor") icon = "🐀";
                if(r.pest === "Mosca") icon = "🪰";
                if(r.pest === "Araña") icon = "🕷️";
                if(r.pest === "Hormiga") icon = "🐜";

                let severityText = r.severity > 1 ? `<div class="badge">Nivel ${r.severity}</div>` : '';

                const html = `
                    <div class="report-item">
                        <div style="font-size: 28px;">${icon}</div>
                        <div style="flex-grow: 1;">
                            <div class="report-date">${r.date}</div>
                            <div class="report-pest">${r.pest}</div>
                            <div class="report-area">${r.area}</div>
                        </div>
                        ${severityText}
                    </div>
                `;
                container.innerHTML += html;
            });
        }
    </script>
